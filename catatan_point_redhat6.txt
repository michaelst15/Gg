@echo off
setlocal enabledelayedexpansion

REM Loop semua file .txt di folder ini
for %%f in (*.txt) do (
    echo Memproses: %%f

    REM Buat file sementara
    > "%%f.tmp" (
        for /f "usebackq delims=" %%l in ("%%f") do (
            set "line=%%l"
            REM Jika baris mengandung "Status            :" maka ganti nilainya
            echo !line! | findstr /c:"Status            :" >nul
            if not errorlevel 1 (
                echo Status            : Pass
            ) else (
                echo !line!
            )
        )
    )

    REM Ganti file asli dengan hasil edit
    move /y "%%f.tmp" "%%f" >nul
)

echo Selesai.
pause
