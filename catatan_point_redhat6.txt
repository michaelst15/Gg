@echo off
setlocal

REM === Loop semua file .txt di folder ini ===
for %%F in (*.txt) do (
    echo Memproses: %%F

    REM Edit langsung isi file dengan PowerShell (in-place)
    powershell -Command ^
        "(Get-Content '%%F') -replace '^Status\s*:.*','Status            : Pass' | Set-Content '%%F'"
)

echo.
echo Selesai. Semua Status sudah diubah jadi 'Pass'.
pause
