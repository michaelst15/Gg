#!/bin/bash

# File hasil output audit/remediasi
output_file="remediation_result.txt"

# Metadata CIS
judul="Judul Audit: 4.2.2 Ensure permissions on all logfiles are configured"
deskripsi="File log di /var/log berisi data penting dari berbagai layanan sistem. Mengatur permission secara ketat mencegah akses tidak sah terhadap informasi sensitif."
nilai_cis="Permission file log harus memastikan:\n- Group tidak memiliki izin write atau execute\n- Other tidak memiliki izin apapun\nPerintah: find /var/log -type f -exec chmod g-wx,o-rwx {} +"

# Timestamp untuk backup
timestamp=$(date +%F-%H%M%S)

# Backup metadata file yang tidak sesuai
log_before="/var/backups/logs_metadata_noncompliant_${timestamp}.txt"
mkdir -p /var/backups

# Backup fisik file yang tidak sesuai ke direktori khusus
backup_dir="/var/backups/logs_noncompliant_backup_${timestamp}"
mkdir -p "$backup_dir"

# Tempat log sementara
log_sementara=$(mktemp)

echo "🔍 Memeriksa file log yang tidak sesuai permission standar..."

find /var/log -type f | while read -r file; do
  perm=$(stat -c "%a" "$file")
  g_write_exec=$(( (perm / 10) % 10 ))
  o_perm=$(( perm % 10 ))

  # Jika grup memiliki write (2) atau execute (1), atau others punya permission
  if (( (g_write_exec & 3) != 0 || o_perm != 0 )); then
    echo "$perm $file" >> "$log_before"
    cp --parents "$file" "$backup_dir" 2>/dev/null
    chmod g-wx,o-rwx "$file"
    echo "✅ Diperbaiki: $file (asal: $perm)" >> "$log_sementara"
  fi
done

modified_count=$(wc -l < "$log_sementara")

# Catat hasil ke file output
{
echo -e "$judul\n"
echo -e "Tindakan Remediasi:\n"
echo -e "1. Backup metadata izin file non-kompatibel: $log_before"
echo -e "2. Backup fisik file log yang dimodifikasi ke: $backup_dir"
echo -e "3. Menjalankan chmod hanya pada file yang tidak sesuai:"
echo -e "   chmod g-wx,o-rwx <file>"
echo -e "4. Total file log yang diperbaiki: $modified_count\n"
echo -e "Daftar file yang diperbaiki:"
cat "$log_sementara"
echo -e "\nDeskripsi:\n$deskripsi"
echo -e "\nNilai CIS:\n$nilai_cis"
echo -e "================================================================================\n"
} >> "$output_file"

# Bersihkan
rm -f "$log_sementara"

echo "✅ Remediasi selesai. Total file diperbaiki: $modified_count"
echo "📄 Metadata backup: $log_before"
echo "📂 Backup file log: $backup_dir"
echo "📝 Log hasil: $output_file"

declare -a file_ungrouped

echo "🔍 Memindai filesystem lokal untuk file yang tidak memiliki grup valid..."

# === Temukan mount point lokal ===
mapfile -t mount_points < <(df --local -P | awk 'NR>1 {print $6}')

# === Temukan semua file dengan grup tidak dikenal (ungrouped) ===

  while IFS= read -r path; do
    file_ungrouped+=("$path")
  done < <(find "$mp" -xdev -nogroup 2>/dev/null)
done

# === Perbaiki semua file yang ditemukan ===
for file in "${file_ungrouped[@]}"; do
  [[ ! -e "$file" && ! -L "$file" ]] && continue

  mount_point=$(df -P "$file" | awk 'NR==2 {print $6}')
  mount_opts=$(mount | grep "on ${mount_point} " | head -n1)

  if echo "$mount_opts" | grep -q '\bro\b'; then
    echo "⚠️  File di filesystem read-only, dilewati: $file" >> "$output_file"
    file_failed+=("$file")
    ((total_failed++))
    continue
  fi

  stat_info=$(stat -c "%U %G %u %g %n" "$file")
  echo "$stat_info" >> "$backup_file"

  if chgrp root "$file"; then
    new_gid=$(stat -c "%g" "$file")
    if [[ "$new_gid" -eq 0 ]]; then
      ((total_fixed++))
    else
      echo "❌ Gagal validasi GID meski chgrp sukses: $file" >> "$output_file"
      file_failed+=("$file")
      ((total_failed++))
    fi
  else

#!/bin/bash

OUTPUT_FILE="audit_results.txt"
USE_PDB=0           # Set ke 1 kalau kamu yakin pakai PDB

echo "🔍 Memeriksa role 'DBA' diberikan kepada user tidak sah..." | tee -a "$OUTPUT_FILE"

# Jalankan query langsung via here-document
RAW_RESULT=$(sqlplus -s / as sysdba <<EOF
SET HEADING OFF
SET FEEDBACK OFF
SET VERIFY OFF
SET PAGESIZE 0
SET LINESIZE 1000
WHENEVER SQLERROR EXIT SQL.SQLCODE

-- Optional switch ke PDB
$( [ "$USE_PDB" -eq 1 ] && echo "ALTER SESSION SET CONTAINER=$PDB_NAME;" )

SELECT 'GRANT:' || GRANTEE FROM DBA_ROLE_PRIVS WHERE GRANTED_ROLE = 'DBA'
UNION
SELECT 'PROXY:' || PROXY || '-' || CLIENT FROM DBA_PROXIES
WHERE CLIENT IN (
  SELECT GRANTEE FROM DBA_ROLE_PRIVS WHERE GRANTED_ROLE = 'DBA'
);

EXIT;
EOF
)

# Tangani error ORA-
echo "$RAW_RESULT" | grep -q "ORA-" && {
  echo "❌ Gagal menjalankan query. Pastikan database/PDB sudah OPEN dan nama PDB benar." | tee -a "$OUTPUT_FILE"
  exit 1
}

# Bersihkan hasil
CLEANED=$(echo "$RAW_RESULT" | tr -d '\r' | sed '/^$/d' | sed 's/^[ \t]*//;s/[ \t]*$//')

AUTHORIZED_USERS="ORACLE ORACLEP"
STATUS="Pass"
VIOLATIONS=""

while IFS= read -r line; do
  case "$line" in
    *:*) ;;
    *) continue ;;
  esac

  GRANTEE=$(echo "$line" | cut -d':' -f2 | cut -d'-' -f1 | tr '[:lower:]' '[:upper:]' | xargs)

  AUTHORIZED=0
  for AUTH in $AUTHORIZED_USERS; do
    if [ "$GRANTEE" = "$AUTH" ]; then
      AUTHORIZED=1
      break
    fi
  done

  if [ "$AUTHORIZED" -ne 1 ]; then
    STATUS="Fail"
    VIOLATIONS="${VIOLATIONS}${line}\n"
  fi
done <<< "$CLEANED"

# Ringkasan hasil
if [ "$STATUS" = "Pass" ]; then
  VALUE="Tidak ditemukan GRANTEE tidak sah dengan role DBA"
else
  VALUE="Ditemukan GRANTEE tidak sah dengan role DBA:\n$VIOLATIONS"
fi

# Simpan hasil ke audit file
{
  echo "Judul Audit : 5.3.4 Ensure 'DBA' Is Revoked from Unauthorized 'GRANTEE'"
  echo "Status : $STATUS"
  printf "Nilai Konfigurasi : $VALUE\n"
  echo "Nilai CIS : 1 (Role DBA hanya boleh diberikan kepada user Oracle internal seperti ORACLE/ORACLEP)"

  echo "-------------------------------------------------------------"
  echo ""
} >> "$OUTPUT_FILE"
